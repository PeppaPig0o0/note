### 网易有道翻译首页爬虫

```python
import urllib.request
'''
使用urllib库中的 request模块中的 urlopen()方法
库->文件夹     模块->py文件    方法->定义的函数
构造一个请求，请求网易有道翻译首页 
'''

# 请求的网址
url = 'https://fanyi.youdao.com/'
# 构造请求
response = urllib.request.urlopen(url)
print(response)
print(type(response))

# 获取请求状态码 status_code
print(response.getcode())
print(response.status)

# 打印响应体中的内容
html = response.read().decode('utf-8')
print(html)
```

### 网易有道翻译内容爬虫

```python
import urllib.request
import urllib.parse
'''
爬取网易有道翻译内容
实现循环爬取多个词语的翻译
实现多语言转换
'''
# 地址,目标
url = 'https://fanyi.youdao.com/translate?smartresult=dict&smartresult=rule'

# 构造了一个头部
header = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/105.0.0.0 Safari/537.36'
}
# 构造一个request对象，看作一个人。url是目标，headers是头
request = urllib.request.Request(url, headers=header)

list1 = ['宫保鸡丁', '锅包肉', '莲汤肉片', '大盘鸡', '香辣蟹', '蒜香小龙虾', '京酱肉丝', '回锅肉', '北京烤鸭']
for word in list1:
    # data是数据，可以认为是身体
    data = {
        'i': word,
        'from': 'AUTO',
        'to': 'AUTO',
        'smartresult': 'dict',
        'client': 'fanyideskweb',
        'salt': '16625225205423',
        'sign': 'a24469b962cc5e64e473cb0f8b2b54de',
        'lts': '1662522520542',
        'bv': '47edca4d7e6ec9bf4fca7156ea36b8ef',
        'doctype': 'json',
        'version': '2.1',
        'keyfrom': 'fanyi.web',
        'action': 'FY_BY_CLICKBUTTION',
    }
    # 把data字典编码成字符串
    data = urllib.parse.urlencode(data).encode('utf-8')
    # 把request这个人和 身体data ，拼在一起。然后再请求目标url
    response = urllib.request.urlopen(request, data=data)
    res = response.read().decode('utf-8')
    print(res)
```
