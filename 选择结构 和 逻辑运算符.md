### 选择结构 和 逻辑运算符

```python
# if 结构
# 一个缩进 一个tab=4个空格
a = 30
if a > 50:
    print('a>50')
print('程序结束')


# if else 结构
score = float(input('请输入你的考试分数：'))
if score >= 60:  # 45分不大于60，所以表达式结果为False
    print('及格')
else:
    print('不及格')
print('程序结束')


# if elif elif elif else 结构
# 一个if结构，可以没有else, 也可以没有elif,但是必须要有if
# 只要一个表达式成立，其他的表达式不再判断
food = input('我叫王珂,我今天晚上想吃：')
if food == '肯德基':
    print('肯德基50块')
elif food == '火锅':
    print('火锅150块')
elif food == '黄焖鸡':  # 成立了 结果是True
    print('黄焖鸡20块')
elif food == '刀削面':
    print('刀削面10块')
elif food == '套餐':
    print('套餐15块')
else:
    print('没你想吃的回去睡吧！')
print('回去睡觉了')

# 逻辑运算符   and与  or或  not非
# and 两边都为真，结果为真
print(10<50 and 20>15)  # True and True -> True
print(10<50 and 20>100)  # True and False -> False
print(100<50 and 20>100)  # False and False -> False

# or 一个为真，结果为真
print(10<50 or 20>15)  # True and True -> True
print(10<50 or 20>100)  # True and False -> True
print(100<50 or 20>100)  # False and False -> False

# not 非真为假 非假为真
print(not 20>10)  # not True  -> False
print(not False)  # True


# 四则运算计算器 加减乘除
a = float(input('请输入一个数字：'))
fuhao = input('请输入运算符号：')
b = float(input('请输入一个数字：'))
if fuhao == '/' and b == 0:
    print('除法分母不能为0')
else:
    if fuhao == '+':
        result = a + b
    elif fuhao == '-':
        result = a - b
    elif fuhao == '*':
        result = a * b
    elif fuhao == '/':
        result = a / b
    elif fuhao == '**':
        result = a ** b
    print(a, fuhao, b, '=', result)


# 判断是否是闰年
# year除以100不能整除  year除以4 可以整除    是闰年
# year除以100能整除 year除以400也可以整除   是闰年
# 2004  是闰年    2000也是闰年    1900年 2100年都不是闰年
year = int(input('请输入年份：'))
if year % 100 == 0:
    if year % 400 == 0:
        print(year, '年是闰年')
    else:
        print(year, '年不是闰年')
else:
    if year % 4 == 0:
        print(year, '年是闰年')
    else:
        print(year, '年不是闰年')


year = int(input('请输入年份：'))
```
